import java.util.Scanner;
public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int a = sc.nextInt(), c = 0; //a변수에 숫자 입력
        for (int i = 0; i < 25; i++) { // n(n+1)/2 시그마 공식을 n을 1씩증가시켜서 입력값 a가 시그마공식 결과값 b보다 작거나 같아지면 i의 값을 c에 넣어주고 break! 
            int b = ((i*(i+1))/2);
            if(a<=b){
                c = i;
                break;
            }
        }
        System.out.print(c + " ");
        while(true){ //a값에 c값을 빼주고 c도 -1씩하여 a가 c보다 작거나 같아지게 만든 후 break! 결과값 출력
            a -= c;
            c--;
            if(a<=c){
                break;
            }
        }
        System.out.println(a);
    }
}
